883 template<unsigned int P> void PWAHBitSet<P>::addFill(bool oneFill, int numBlocks, int firstBlockIndex){
884  const bool DEBUGGING = false;
885  if (_VERIFY) assert(numBlocks > 0);
886  if (DEBUGGING) cout << "PWAHBitSet::addFill(" << (oneFill ? "1" : "0") << ", " << numBlocks << " blocks). firstBlockIndex=" << firstBlockIndex << endl;
887 
888  // Check last compressed word
889  if (_words.size() > 0){
890  // Starting with the last used partition, scan all partitions to identify a possibly
891  // existing (extended or simple) fill of the same type
892  short prevFillHeadPartition = -1;
893  for (short i = _lastUsedPartition; i >= 0; i--){
894  if (_words.size() == 1 && i == 0) continue; // w = 0, i = 0 ==> plain block
895 
896  if ((oneFill && is_onefill(_words.back(), i)) || (!oneFill && is_zerofill(_words.back(), i))){
897  // Partition is a fill of same type as fill being added!
898  prevFillHeadPartition = i;
899  } else {
900  // Partition is not a fill, or not of the same type as the fill currently being added
901  break;
902  }
903  }
904 
905  if (prevFillHeadPartition != -1){
906  const int existingFillLength = fill_length(_words.back(), prevFillHeadPartition);
907  firstBlockIndex -= existingFillLength;
908  numBlocks = existingFillLength + numBlocks;
909  if (DEBUGGING) cout << "PWAHBitSet::addFill(" << (oneFill ? "1" : "0") << ")" << "-- Encountered fill of size " << fill_length(_words.back(), prevFillHeadPartition) << " blocks at partition " << _lastUsedPartition << "(spanning " << (_lastUsedPartition - prevFillHeadPartition + 1) << " partition(s)), extending fill to size " << numBlocks << "..." << endl;
910 
911  // Decrease _lastUsedPartition, call to 'addPartition' below will increase _lastUsedPartition
912  if (prevFillHeadPartition == 0){
913  _words.pop_back();
914  _lastUsedPartition = P - 1;
915  } else {
916  // Clear the partition(s) containing the fill that is about to be extended
917  for (short i = prevFillHeadPartition; i <= _lastUsedPartition; i++){
918  if (DEBUGGING) cout << "PWAHBitSet::addFill -- clearing partition " << i << endl;
919  _words[_words.size() - 1] = clear_partition(_words.back(), i);
920  }
921 
922  if (DEBUGGING) cout << "PWAHBitSet::addFill -- result after clearing partition(s): " << toBitString(_words.back()) << endl;
923  // Decrease counter, that (cleared) partition will be overwritten
924  _lastUsedPartition = prevFillHeadPartition - 1;
925  }
926  } // else: last partitions do not represent a fill or not a fill of equal type
927  } // else: no compressed words available
928 
929  if (numBlocks > _maxBlocksPerFill && _lastUsedPartition == P - 2){
930  if (DEBUGGING) cout << "PWAHBitSet::addFill -- extended fill requested, but at last partition of current word" << endl;

932  // Store the first part of the fill in the last partition of this word, the second part
933  // of the fill should go into the next word. When necessary, the second part can be stored
934  // as an extended fill (see below).
935 
936  // Store _maxBlocksPerFill of numBlocks in the last partition of this word
937  addFillPartition(oneFill, _maxBlocksPerFill, firstBlockIndex);
938 
939  firstBlockIndex += _maxBlocksPerFill;
940  numBlocks -= _maxBlocksPerFill;
941  }
942 
943  if (numBlocks > _maxBlocksPerFill){
944  // Add extended fill.
945  if (DEBUGGING) cout << "PWAHBitSet::addFill -- adding extended fill of " << numBlocks << " blocks..." << endl;
946  addExtendedFillPartitions(oneFill, numBlocks, firstBlockIndex);
947  } else {
948  // Simply add new fill
949  if (DEBUGGING) cout << "PWAHBitSet::addFill -- adding simple fill of " << numBlocks << " blocks..." << endl;
950  addFillPartition(oneFill, numBlocks, firstBlockIndex);
951  }
952 
953  if (DEBUGGING) cout << "PWAHBitSet::addFill -- resulting last word: " << toBitString(_words.back()) << endl;
954 }
