604  // Totally out of bounds. Note that the '>' in stead of '>=' is intentional!
605  // Skip this BitSet
606  if (debug) cout << "BitSet " << i << ": totally out of bounds (" << sWordIndex[i] << " > " << bitSets[i]->_compressedBits.size() << "), skipping..." << endl;
607  continue;
608  }
609 
610  if (debug) cout << "Aligning BitSet " << i << ", currently at block " << sBlockIndex[i] << ", should be at block " << rBlockIndex << endl;
611  while (sBlockIndex[i] < rBlockIndex){
612  if (debug) cout << "BitSet " << i << ": at block " << sBlockIndex[i] << endl;
613 
614  // Perform some aligning: this BitSet is not yet at block index rBlockIndex
615  if (sWordIndex[i] >= bitSets[i]->_compressedBits.size()){
616  // Out of bounds of compressed bits, no more skipping
617  if (debug) cout << "BitSet " << i << ": running out of bounds, breaking alignment..." << endl;
618  break;
619  }
620 
621  currWord = bitSets[i]->_compressedBits[sWordIndex[i]];
622 
623  if (IS_FILL(currWord)){
624  // Fill word. Check length and skip accordingly
625  currFillLengthRemaining = FILL_LENGTH(currWord) - sWordOffset[i];
626  if (currFillLengthRemaining <= rBlockIndex - sBlockIndex[i]){
627  // Skip entire length
628  if (debug) cout << "BitSet " << i << ": skipping one entire fill word of length " << currFillLengthRemaining << endl;
629  sWordIndex[i]++;
630  sWordOffset[i] = 0;
631  sBlockIndex[i] += currFillLengthRemaining;
632  } else {
633  // Can't skip the entire length...
634  if (debug) cout << "BitSet " << i << ": skipping part of fill word: only " << (rBlockIndex - sBlockIndex[i]) << " of " << currFillLengthRemaining << " blocks" << endl;
635  sWordOffset[i] += rBlockIndex - sBlockIndex[i];
636  sBlockIndex[i] += rBlockIndex - sBlockIndex[i];
637  }
638  } else {
639  // Literal, skip one block
640  if (debug) cout << "BitSet " << i << ": skipping one literal word (one block)" << endl;
641  sBlockIndex[i]++;
642  sWordIndex[i]++;
643  }
644  } // end while: skipping done
645  if (rBlockIndex > sBlockIndex[i]){
646  if (debug) cout << "Alignment of BitSet " << i << " failed, skipping BitSet..." << endl;
647  continue;
648  }
649  if (debug) cout << "Done aligning BitSet " << i << ", now at word " << sWordIndex[i] << endl;
650 
651  if (sWordIndex[i] < bitSets[i]->_compressedBits.size()){
652  // Within bounds
653  if (debug) cout << "Current compressed word of bitset " << i << ": " << toBitString(bitSets[i]->_compressedBits[sWordIndex[i]]) << endl;
654  currWord = bitSets[i]->_compressedBits[sWordIndex[i]];
655  } else if (sWordIndex[i] == bitSets[i]->_compressedBits.size()){
656  // One out of bounds, take plain word
657  if (debug) cout << "Considering plain word of bitset " << i << ": " << toBitString(bitSets[i]->_plainWord) << endl;
658  //sBlockIndex[i] = bitSets[i]->_plainWordBlockSeq;
659  currWord = bitSets[i]->_plainWord;
660  } else {
661  // Totally out of bounds, skip this BitSet
662  if (debug) cout << "BitSet " << i << ": totally out of bounds after skipping" << endl;
663  continue;
664  }
665 
666  if (IS_ONEFILL(currWord)){
667  currFillLengthRemaining = FILL_LENGTH(currWord) - sWordOffset[i];
668  if (currFillLengthRemaining > largestOneFillSize){
669  largestOneFill = i;
670  largestOneFillSize = currFillLengthRemaining;
671  }
672  } else if (IS_ZEROFILL(currWord)){
673  // Zero fills are boring. Skip the 0-fill, move on to the next word of this BitSet
674  currFillLengthRemaining = FILL_LENGTH(currWord) - sWordOffset[i];
675 
676  if (currFillLengthRemaining < shortestZeroFillSize || shortestZeroFillSize == -1){
677  // This 0-fill is the shortest one
678  shortestZeroFillSize = currFillLengthRemaining;
679  }
680 
681  sWordIndex[i]++;
682  sWordOffset[i] = 0;
683  sBlockIndex[i] += currFillLengthRemaining;
684  if (debug) cout << "BitSet " << i << ": 0-fill of length " << currFillLengthRemaining << " detected, skipping to blockindex " << sBlockIndex[i] << "..." << endl;
685 
686  } else {
687  // Literal
688  if (largestOneFillSize == -1){
689  if (debug) cout << "BitSet " << i << ": merging literal word in resulting literal" << endl;
690  currMergedLiteral |= currWord;
691  }
692  sBlockIndex[i]++;
693  sWordIndex[i]++;
694  }
695  } // end for
696  if (debug) cout << "Done preprocessing " << numBitSets << " source BitSets, writing word to result" << endl;
697 
698  // Perform optimal action
699  if (largestOneFillSize != -1){
700  // Add 1-fill
701  currWord = bitSets[largestOneFill]->_compressedBits[sWordIndex[largestOneFill]];
702 
703  if (debug) cout << "Adding 1-fill of size " << largestOneFillSize << " to result (blockindex=" << rBlockIndex << ")" <<endl;
704  result->addOneFill(largestOneFillSize);
705  rBlockIndex += largestOneFillSize;
706 
707  // Increase position in BitSet which was just used
708  sBlockIndex[largestOneFill] += largestOneFillSize;
709  sWordIndex[largestOneFill]++;
710  sWordOffset[largestOneFill] = 0;
711  } else if (currMergedLiteral != 0){
712  // Add literal
713  if (debug) cout << "Adding literal word to result (blockindex=" << rBlockIndex << "): " << toBitString(currMergedLiteral) << endl;
714  result->addLiteral(currMergedLiteral);
715  rBlockIndex++;
716  } else if (shortestZeroFillSize != -1){
717  // All BitSets are currently showing a 0-fill. Add a zero fill of length minimal length,
718  // and jump to the first possible block for which at least one BitSet shows a literal
719  // or a 1-fill.
720 
721  // Add zero fill
722  if (debug) cout << "Adding 0-fill of size " << shortestZeroFillSize << " to result (blockindex=" << rBlockIndex << ")" <<endl;
723 
724  result->addZeroFill(shortestZeroFillSize);
725  rBlockIndex += shortestZeroFillSize;
726  } else {
727  // Nothing more to do!
728  if (debug) cout << "Nothing to write to result, merging done!" << endl;
729  break;
730  }
731  } // end while
732 
733  if (debug) cout << "Cleaning up..." << endl;
734  delete[] sWordIndex;
735  delete[] sWordOffset;
736  delete[] sBlockIndex;
737 
738  // Decompress last word of resulting BitSet to plain word
739  result->_lastBitIndex = lastBitIndex;
740  if (result->_compressedBits.size() > 0) result->decompressLastWord();
741 
742  if (debug) cout << "done multiway!" << endl;
743 }

745 void WAHBitSet::decompressLastWord(){
746  if (this->_compressedBits.size() == 0) throw string ("Can't decompress last word, BitSet has no compressed words!");
747  if (this->_plainWord != 0) throw string ("Can't decompress last word, plain word != 0");
748 
749  if (IS_FILL(this->_compressedBits.back())){
750  unsigned int fillLength = FILL_LENGTH(this->_compressedBits.back());
751 
752  if (IS_ONEFILL(this->_compressedBits.back())){
753  this->_plainWord = LITERAL_ONEFILL;
754  this->_compressedBits.pop_back();
755 
756  if (fillLength >= 2) _compressedBits.push_back((fillLength - 1) | EMPTY_ONEFILL);
757  } else {
758  this->_plainWord = LITERAL_ZEROFILL;
759  this->_compressedBits.pop_back();
760  if (fillLength >= 2) _compressedBits.push_back((fillLength - 1) | EMPTY_ZEROFILL);
761  }
762  } else {
763  this->_plainWord = this->_compressedBits.back();
764  this->_compressedBits.pop_back();
765  }
766 
767  _plainWordBlockSeq--;
768 }
