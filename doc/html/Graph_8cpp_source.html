51 double Graph::computeLocalClusteringCoefficient(int vertexIndex){
52  StaticBitSet neighbourhood = StaticBitSet(_vertices.size());
53  int neighbourhoodSize = 0;
54 
55  // Populate neighbourhood
56  for (unsigned int i = 0; i < _vertices[vertexIndex].size(); i++){
57  if (_vertices[vertexIndex][i] == vertexIndex) continue; // ignore self-loops
58  if (neighbourhood.get(_vertices[vertexIndex][i])) continue; // ignore duplicate edges
59 
60  neighbourhood.set(_vertices[vertexIndex][i]);
61  neighbourhoodSize++;
62  }
63 
64  if (neighbourhoodSize > 0){
65  // Loop over adjacent vertices a_i and check whether a_i is connected
66  // to an other adjacent vertex a_j
67  int currNeighbour;
68  int numEdgesInNeighbourhood = 0;
69  for (unsigned int i = 0; i < _vertices[vertexIndex].size(); i++){
70  if (_vertices[vertexIndex][i] == vertexIndex) continue; // ignore self-loops
71  if (!neighbourhood.get(_vertices[vertexIndex][i])) continue; // ignore duplicate edges
72 
73  currNeighbour = _vertices[vertexIndex][i];
74 
75  // Loop over adjacent vertices of the neighbour
76  for (unsigned int j = 0; j < _vertices[currNeighbour].size(); j++){
77  if (neighbourhood.get(_vertices[currNeighbour][j])){
78  numEdgesInNeighbourhood++;
79  }
80  }
81  }
82 
83  double localC = (double) numEdgesInNeighbourhood / (neighbourhoodSize * neighbourhoodSize);
84  if (localC > 1){
85  cerr << "vertex " << vertexIndex << " local clustering coefficient=" << localC << ", numEdgesInNeighbourhood=" << numEdgesInNeighbourhood << ", neighbourhoodSize=" << neighbourhoodSize << endl;
86  assert(localC <= 1);
87  }
