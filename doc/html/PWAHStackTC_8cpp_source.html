436 template<class B> bool PWAHStackTC<B>::componentHasSelfLoop(int componentIndex){
437  if (_reflexitive || _storeComponentVertices){
438  // Self-loops were not stored explicitly, since their existence can be shown
439  // implicitly.
440 
441  if (!_reflexitive){
442  // Some components may have self-loops, others not
443  if (_componentSizes[componentIndex] > 1){
444  // Self-loop implied
445  return true;
446  } else {
447  // Component contains only one vertex
448  if (_vertexSelfLoop->get(_componentVertices[componentIndex][0])){
449  // That vertex has a self-loop
450  return true;
451  } // else: vertex doesn't have a a self-loop, therefore, the component has no self-loop
452  }
453  } else {
454  // Reflexitive transitive closure: all components have self-loops
455  return true;
456  }
457  } else {
458  // Self-loops were stored explicitly, this function should not be used
459  throw string("PWAHStackTC::componentHasSelfLoop can not be used when self-loops are stored explicitly!");
460  }
461  return false;

543 template<> void PWAHStackTC<PWAHBitSet<8> >::reachablepairs(vector<unsigned int>& sources, vector<unsigned int>& targets, vector<vector<unsigned int> >& reachable){
544  if (sources.size() == 0 || targets.size() == 0) return;
545  reachable.reserve(sources.size());
546 
547 
548  // First, find out the strongly connected components of the target vertices
549

556

558  // return the vertex indices, rather than the component indices.

560  const unsigned int target_vertex_index = *it_tgt;

563

565 
566  if (target_component_vertices[target_comp_index] == NULL){
567  target_component_vertices[target_comp_index] = new vector<unsigned int>();
568  target_components.push_back(target_comp_index);
569  }
570  target_component_vertices[target_comp_index]->push_back(target_vertex_index);
571  }
572

574  sort(target_components.begin(), target_components.end());
575 
576  // Stores for each source component whether we've stored its result in the 'reachable' vector already. This
577  // happens when multiple source vertices share the same component.
578  vector<unsigned int> components_in_reachable = vector<unsigned int>(_componentSuccessors.size(), UINT_MAX);
579 
580  for (unsigned int i = 0; i < sources.size(); i++){
581  const unsigned int src_vertex_index = sources[i];
582 
583  if (src_vertex_index >= _graph->getNumberOfVertices()) throw range_error("Source index out of bounds");
584  const unsigned int src_comp_index = _vertexComponents[src_vertex_index];
585 
586  if (components_in_reachable[src_comp_index] != UINT_MAX){
587  // We already have a result for this source component (i.e., we've seen a source vertex from this
588  // strongly connected component before). No need to iterate over the bitset, just copy
589  // the previous result
590  reachable.push_back(reachable[components_in_reachable[src_comp_index]]);
591  } else {
592  // Iterate over bit vectors to see which target components (and vertices) are reachable
593  reachable.push_back(vector<unsigned int>());
594  components_in_reachable[src_comp_index] = i;
595 
596  if (_componentSuccessors[src_comp_index] == NULL){
597  // This component can't reach anything - leave result vector empty
598  continue;
599  }
600 
601  // Implementation of a single scan over a compressed bitset using the 'incr_get' method of
602  // PWAHBitSet. This method requires that the queries to the PWAHBitSet are in increasing order
603  // of bit index, but then guarantees only a single pass is made over the data structure, which
604  // is of course much more efficient than scanning over the entire PWAHBitSet for every single
605  // query.
606  for (unsigned int j = 0; j < target_components.size(); j++){
607  const unsigned int curr_target_component_index = target_components[j];
608 
609  bool target_reachable = false;
610  bool decided = false;
611 
612  if (src_comp_index == curr_target_component_index){
613  // Source and target lie in the same component. If the component has size > 1, then any
614  // target vertex can be reached from any source within that same component. If the component
615  // consists of a single vertex, then reachability requires:
616  // 1) a self-loop OR
617  // 2) a reflexive transitive closure
618 
619  if (_reflexitive || _componentSizes[src_comp_index] > 1){
620  // Always reachable
621  target_reachable = true;
622  decided = true;
623  } else if (_storeComponentVertices) {
624  // Self-loops were not stored explicitly, since their existence can be shown implicitly.
625  target_reachable = this->componentHasSelfLoop(src_comp_index);
626  decided = true;
627  } else {
628  // Self-loops were stored explicitly
629  decided = false;
630  }
631  }
632 
633  if (!decided){
634  target_reachable = _componentSuccessors[src_comp_index]->incr_get(curr_target_component_index);
635  decided = true;
636  }
637 
638  if (target_reachable){
639  // add all target vertices of that component to the list of reachable vertices.
640  reachable[i].insert(reachable[i].end(),
641  target_component_vertices[curr_target_component_index]->begin(),
642  target_component_vertices[curr_target_component_index]->end()
643  );
644  }
645  }
646  }
647  }
648 
649  // Be nice and free some memory
650  for (unsigned int i = 0; i < target_component_vertices.size(); i++){
651  if (target_component_vertices[i] != NULL) delete target_component_vertices[i];
652  }
653 }

658 template<class B> bool PWAHStackTC<B>::reachable(unsigned int src, unsigned int dst){
659  const bool DEBUGGING = false;
660  if (src >= _graph->getNumberOfVertices()) throw range_error("Source index out of bounds");
661  if (dst >= _graph->getNumberOfVertices()) throw range_error("Source index out of bounds");
662  if (DEBUGGING) cout << "PWAHStackTC::reachable " << src << " -> " << dst << " (vertex indices)" << endl;
663

665  int dstComponent = _vertexComponents[dst];
666 
667  if (DEBUGGING) cout << "PWAHStackTC::reachable " << srcComponent << " -> " << dstComponent << " (component indices)" << endl;
668 
669  if (_componentSuccessors[srcComponent] == NULL){
670  // Source component didn't have any adjacent components
671  return false;
672  } else if (srcComponent == dstComponent){
673  // Source and destination are within the same component!
674  // Determine whether the component contains a self-loop
675  if (_reflexitive || _componentSizes[srcComponent] > 1){
676  return true;
677  } else if (_componentSizes[srcComponent] == 1 && _storeComponentVertices){
678  return _vertexSelfLoop->get(src);
679  } // else: unable to determine which vertex is the only member of the component...
680  }
681 
682  return _componentSuccessors[srcComponent]->get(dstComponent);

797 template<class B> double PWAHStackTC<B>::computeLocalClusteringCoefficient(int componentIndex){
798  if (_componentSuccessors[componentIndex] == NULL) return 0;
799  StaticBitSet neighbourhood = StaticBitSet(_componentSuccessors.size());
800  int neighbourhoodSize = 0;
801 
802  // Register neighbourhood in StaticBitSet
803  BitSetIterator* it = _componentSuccessors[componentIndex]->iterator();
804  int currNeighbourIndex = it->next();
805  while (it->next() >= 0){
806  if (currNeighbourIndex != componentIndex){ // skip self-loops
807  cout << "component " << componentIndex << ": " << currNeighbourIndex << " is neighbour" << endl;
808  neighbourhood.set(currNeighbourIndex);
809  neighbourhoodSize++;
810  }
811  currNeighbourIndex = it->next();
812  }
813  delete it;
814  cout << "Neighbourhood of component " << componentIndex << " contains " << neighbourhoodSize << " components" << endl;
815  if(neighbourhoodSize == 0) return 0;
816 
817  int numEdgesInNeighbourhood = 0;
818  int currNeighbourNeighbourIndex;
819  it = _componentSuccessors[componentIndex]->iterator();
820  BitSetIterator* nIt;
821  currNeighbourIndex = it->next();
822  while (it->next() >= 0){
823  // Process currSuccessorIndex, check whether it neighbours are also neighbours
824  // of componentIndex
825 
826  if (_componentSuccessors[currNeighbourIndex] != NULL){
827  cout << "Neighbourhood of " << componentIndex << ": iterating over neighbours of " << currNeighbourIndex << endl;
828  nIt = _componentSuccessors[currNeighbourIndex]->iterator();
829  currNeighbourNeighbourIndex = nIt->next();
830 
831  while (currNeighbourNeighbourIndex >= 0){
832  if (neighbourhood.get(currNeighbourNeighbourIndex)){
833  cout << "Neighbourhood of " << componentIndex << ": " << currNeighbourIndex << " has neighbour " << currNeighbourNeighbourIndex << ", which is also a neighbour of " << componentIndex << endl;
834  numEdgesInNeighbourhood++;
835  }
836  currNeighbourNeighbourIndex = nIt->next();
837  }
838  delete nIt;
839  }
840 
841  currNeighbourIndex = it->next();
842  }
843  delete it;
844 
845  double res = (double) numEdgesInNeighbourhood / (neighbourhoodSize * neighbourhoodSize);
846  cout << componentIndex << " has local correlation coefficient of " << res << endl;
847  assert(res <= 1);
848  return res;
849 }
